const express = require("express"); const { join } = require("path"); const app = express(); var router = express.Router(); router.get('/', function(req, res){ res.render('index', { title: 'Home' }); }); router.get('/level1', function(req, res){ res.render('level1', { title: 'Level 1' }); }); router.get('/level2', function(req, res){ res.render('level2', { title: 'Level 2' }); }); module.exports = router; // Serve static assets from the /public folder app.use(express.static(join(__dirname, "public"))); // Endpoint to serve the configuration file app.get("/auth_config.json", (req, res) => { res.sendFile(join(__dirname, "auth_config.json")); }); // Serve the index page for all other requests app.get("/*", (_, res) => { res.sendFile(join(__dirname, "index.html")); }); // Listen on port 3000 app.listen(3000, () => console.log("Application running on port 3000"));
